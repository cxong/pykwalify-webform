      enum:
      - commercial

      enum:
      - complete
      - very active
      - active
      - sporadic
      - halted
      type: str
    feed:
      desc: URL of RSS/Atom feed for the game's news, or releases
      type: str
    framework:
      desc: Framework(s) or engine(s) used for the game, e.g. SDL2
      sequence:
      - type: str
        unique: true
      type: seq
    images:
      desc: URLs of screenshots for the game
      sequence:
      - type: str
        unique: true
      type: seq
    info:
      desc: Any notable information about the game
      type: str
    lang:
      desc: Main programming language(s) used for the game, e.g. C++
      sequence:
      - type: str
        unique: true
      type: seq
    license:
      required: true
      sequence:
      - enum: &id001
        - Custom
        - Multiple
        - PD
        - As-is
        - AFL3
        - AGPL3
        - Apache
        - Artistic
        - BSD
        - BSD2
        - BSD4
        - bzip2
        - CC0
        - CC-BY
        - CC-BY-NC
        - CC-BY-NC-ND
        - CC-BY-NC-SA
        - CC-BY-SA
        - GPL1
        - GPL2
        - GPL3
        - IJG
        - ISC
        - JRL
        - LGPL2
        - LGPL3
        - Libpng
        - MAME
        - MIT
        - MPL
        - MS-PL
        - NGPL
        - OFL
        - WTFPL
        - Zlib
        include: licenses
        type: str
        unique: true
      type: seq
    multiplayer:
      sequence:
      - enum:
        - Competitive
        - Local
        - Online
        - LAN
        - Split-screen
        - Co-op
        - Hotseat
        - Matchmaking
        type: str
        unique: true
      type: seq
    name:
      desc: Name of the clone
      required: true
      type: str
    originals:
      desc: Name(s) of games this clones
      required: true
      sequence:
      - type: str
        unique: true
      type: seq
    repo:
      desc: URL where the source code can be found; e.g. a GitHub project page
      required: false
      type: str

      enum:
      - playable
      - semi-playable
      - unplayable
      type: str
    type:
      enum:
      - remake
      - clone
      - similar
      - tool
      required: true
      type: str
    updated:
      desc: Use today's date
      format: '%Y-%m-%d'
      type: date
    url:
      desc: URL of the game's main page (if any; should be different from repo)
      type: str
    video:

        indiedb:

          type: int

          desc: Numeric video ID; copy the embed code, look for e.g. https://www.moddb.com/media/iframe/535281,
            use "535281"
          type: int
        vimeo:
          desc: Numeric video ID e.g. for https://vimeo.com/10245458, use "10245458"
          type: int
        youtube:
          desc: 11-char video ID e.g. for https://www.youtube.com/watch?v=DowgWKtGAD8,
            use "DowgWKtGAD8"
          range:
            max: 11
            min: 11
          type: str
      type: map
  type: map
